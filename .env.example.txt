# ==========================================

# --- API 密钥配置 ---
# 即梦 API Session ID (从浏览器 Cookie 获取)
JIMENG_SESSION_ID=your_session_id_here

# Gemini API Key (https://aistudio.google.com/app/apikey)
GEMINI_API_KEY=your_gemini_api_key_here

# --- 模型选择 ---
# 智谱 API 配置（推荐，用于文案二创）
# 获取 API Key: https://open.bigmodel.cn/usercenter/apikeys
ZHIPU_API_KEY=your_zhipu_api_key_here
ZHIPU_BASE_URL=https://open.bigmodel.cn/api/paas/v4
ZHIPU_MODEL=glm-4-flash

# Poe API 配置 (用于场景拆分和生图提示词生成，可选)
# 获取 API Key: https://poe.com/api_key
POE_API_KEY=your_poe_api_key_here
POE_BASE_URL=https://api.poe.com/v1
# 可用的 Bot: Claude-3.5-Sonnet, GPT-4, Claude-Opus, Claude-Instant
POE_BOT_NAME=Claude-3.5-Sonnet

# Coze API 配置（播客音频生成备选方案，可选）
# 用于替代 SoulX-Podcast 生成播客音频
COZE_API_TOKEN=your_coze_api_token_here
COZE_MCP_URL=https://mcp.coze.cn/v1/plugins/7537547135328419903

# Gemini 模型 [已弃用，改用 Poe API]
# 可选值: gemini-2.0-pro-exp-02-05, gemini-1.5-flash, gemini-1.5-pro
GEMINI_MODEL=gemini-2.0-pro-exp-02-05

# --- 即梦生图参数 ---
# 生图模型
JIMENG_MODEL=jimeng-4.0
# 画幅比例 (根据你的要求默认为 4:3)
JIMENG_RATIO=4:3
# 分辨率 (可选: 540p, 720p, 1080p, 2k)
JIMENG_RESOLUTION=2k

# --- DeepSeek API 配置 (用于字幕优化) ---
# 蓝云 MaaS API Key (https://maas-api.lanyun.net/)
DEEPSEEK_API_KEY=your_deepseek_api_key_here
# 蓝云 MaaS API 基础 URL
DEEPSEEK_BASE_URL=https://maas-api.lanyun.net/v1
# DeepSeek 模型名称
DEEPSEEK_MODEL=/maas/deepseek-ai/DeepSeek-V3.2

# --- 其他配置 ---
# 运行环境 (可选: dev, prod)

# 情感 (neutral/gentle/cheerful/serious)

# --- SoulX-Podcast 配置 (播客音频生成) ---
